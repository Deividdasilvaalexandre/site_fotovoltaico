int nivel = 0;

void setup()
{
  pinMode(A5, INPUT);
  Serial.begin(9600);
  pinMode(2, OUTPUT);
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(5, OUTPUT);
  pinMode(6, OUTPUT);
  pinMode(7, OUTPUT);
  pinMode(8, OUTPUT);

  nivel = analogRead(A5);
}

void loop()

  Serial.println(nivel);
  if (nivel >= 0 && nivel <= 99) {

    digitalWrite(8, LOW);

    digitalWrite(6, LOW);
